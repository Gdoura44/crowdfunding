{
  "_id": ObjectId,
  
  // actorId + actorRole : seule référence autoritative de l’auteur de l’action
  // (le champ adminId a été retiré pour éviter la redondance).
  "actorId": { "type": "ObjectId", "ref": "User", "required": true },
  "actorRole": {
    "type": "string",
    "enum": ["USER", "ADMIN"],
    "required": true
  },
  
  "action": { "type": "string", "required": true },
  // exemples d’actions :
  //   "VALIDATE_PROJECT", "PUBLISH_PROJECT", "REVOKE_APPROVAL",
  //   "DEACTIVATE_USER", "REACTIVATE_USER", "RETRY_ANALYSIS",
  //   "RESOLVE_REPORT", "APPROVE_PAYOUT", "CANCEL_PAYOUT",
  //   "REACTIVATE_PROJECT", "ARCHIVE_PROJECT", "DELETE_USER",
  //   "SUBMIT_PROJECT", "CREATE_INVESTMENT", "CANCEL_INVESTMENT"
  
  "targetType": { "type": "string", "required": true },
  // exemples de targetType :
  //   "Project", "User", "Report", "Payout", "Investment", "Comment",
  //   "FailedWorkflowEvent", "FailedCancellationEvent",
  //   "FailedRefundEvent", "FailedPayoutEvent"
  
  "targetId": { "type": "ObjectId", "required": true },
  
  "details": { "type": "Object" },
  // details : contexte additionnel, par exemple :
  //   - motif de décision (ex. "Contenu spam détecté")
  //   - transition de statut (ex. { from: "PENDING", to: "APPROVED" })
  //   - notes de validation, motif de résolution, mise à jour RIB, etc.
  
  "timestamp": { "type": "Date", "default": "Date.now", "required": true }
}

// Indexes
db.auditLogs.createIndex({ actorId: 1, timestamp: -1 });
db.auditLogs.createIndex({ targetId: 1, targetType: 1 });
db.auditLogs.createIndex({ timestamp: -1 });
db.auditLogs.createIndex({ action: 1, timestamp: -1 });

// ============================================================================
// LIGNES DIRECTRICES — UTILISATION DE L’AUDIT LOG
// ============================================================================
//
// **Quand créer un audit log :**
// - Toutes les actions admin (validation projet, désactivation utilisateur, résolution
//   de signalement, annulation d’approbation, etc.)
// - Toutes les opérations de relance (analyse IA, remboursement, annulation, payout)
// - Actions sensibles utilisateur (suppression de compte, mise à jour de RIB, etc.)
// - Opérations financières (validation/annulation payout, création d’investissement)
// - Transitions de statut critiques (publication, suspension, réactivation projet)
//
// **À ne pas logger :**
// - Lectures (consultation de projets, recherche, récupération de notifications)
// - Navigation utilisateur normale
// - Actions automatisées (cron) sauf si elles modifient un état critique
//   (exception : l’expiration d’un projet DOIT être loggée).
//
// **Référence après suppression d’un utilisateur :**
// - actorId reste valide même après suppression (pointe vers le document anonymisé).
// - actorRole conservé pour indiquer si l’action était user/admin.
// - Permet les audits de conformité même après suppression de compte.
//
// ============================================================================
// EXEMPLES D’ENTRÉES
// ============================================================================
//
// **Admin valide un projet :**
// {
//   actorId: ObjectId("admin123"),
//   actorRole: "ADMIN",
//   action: "VALIDATE_PROJECT",
//   targetType: "Project",
//   targetId: ObjectId("proj456"),
//   details: {
//     decision: "APPROVED",
//     notes: "Plan d’affaires solide, objectif réaliste",
//     previousStatus: "UNDER_REVIEW",
//     newStatus: "APPROVED"
//   },
//   timestamp: ISODate("2026-04-15T14:30:00Z")
// }
//
// **Admin annule l’approbation d’un projet :**
// {
//   actorId: ObjectId("admin123"),
//   actorRole: "ADMIN",
//   action: "REVOKE_APPROVAL",
//   targetType: "Project",
//   targetId: ObjectId("proj456"),
//   details: { reason: "Description incomplète", previousStatus: "APPROVED", newStatus: "REJECTED" },
//   timestamp: ISODate("2026-04-15T15:00:00Z")
// }
//
// **Utilisateur annule un investissement :**
// {
//   actorId: ObjectId("user789"),
//   actorRole: "USER",
//   action: "CANCEL_INVESTMENT",
//   targetType: "Investment",
//   targetId: ObjectId("inv321"),
//   details: { amount: 500, reason: "USER_REQUESTED", projectTitle: "Initiative panneaux solaires" },
//   timestamp: ISODate("2026-04-15T15:45:00Z")
// }
//
// **Admin relance un payout en échec :**
// {
//   actorId: ObjectId("admin123"),
//   actorRole: "ADMIN",
//   action: "RETRY_PAYOUT",
//   targetType: "Payout",
//   targetId: ObjectId("payout555"),
//   details: { previousStatus: "FAILED", newStatus: "COMPLETED", retryCount: 2,
//              failureReason: "IBAN invalide — corrigé par admin", amount: 10000 },
//   timestamp: ISODate("2026-04-15T16:00:00Z")
// }
//
// **Cron expire un projet :**
// {
//   actorId: ObjectId("system_cron_user"),
//   actorRole: "ADMIN",  // les actions système utilisent le rôle ADMIN
//   action: "EXPIRE_PROJECT",
//   targetType: "Project",
//   targetId: ObjectId("proj789"),
//   details: { deadline: ISODate("2026-04-15T00:00:00Z"), currentFunding: 5000,
//              fundingGoal: 10000, refundsInitiated: 12 },
//   timestamp: ISODate("2026-04-15T01:00:00Z")
// }
//
// ============================================================================
// CONFORMITÉ & RÉTENTION
// ============================================================================
//
// **Rétention :** Les audit logs DOIVENT être conservés au minimum 7 ans (conformité financière).
// Ne JAMAIS supprimer un audit log même si les entités liées sont supprimées.
// La suppression d’un utilisateur anonymise l’actorId mais préserve le log.
//
// **Contrôle d’accès :**
// - Seul le rôle ADMIN peut consulter les audit logs.
// - La consultation des audit logs n’est PAS elle-même loggée (éviter la récursion).
// - Export possible pour conformité (JSON/CSV).
//
// **Données sensibles :**
// - NE PAS logger de PII dans details (e-mails, mots de passe, RIB en clair, etc.).
// - NE PAS logger les payloads complets (seulement les champs pertinents).
// - L’action seule suffit à identifier ce qui a changé (ex. "UPDATE_BANK_DETAILS").
//
// ============================================================================
// UTILISATEUR SYSTÈME (CRON)
// ============================================================================
// Un utilisateur système dédié est créé à l’initialisation de la base :
//   _id: ObjectId("000000000000000000000001")  (ou un identifiant réservé fixe)
//   email: "system@crowdfunding.local"
//   role: "ADMIN"
//   isActive: true
//   (passwordHash et tokens null/empty — jamais utilisé pour se connecter)
//
// Cet utilisateur est l’actorId de toutes les actions cron / système. L’ID est codé en dur.
// ============================================================================
