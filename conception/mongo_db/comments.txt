  "_id": ObjectId,
  "projectId": { "type": "ObjectId", "ref": "Project", "index": true, "required": true },
  "userId": { "type": "ObjectId", "ref": "User", "index": true, "required": true },
  "authorLabel": "string",
  "content": { "type": "string", "required": true, "maxLength": 1000 },

  // Modération (admin)
  "isHidden": { "type": "boolean", "default": false, "index": true },
  "hiddenReason": "string",
  "hiddenAt": "Date",
  "hiddenBy": { "type": "ObjectId", "ref": "User" },

  // Suppression douce (auteur ou admin)
  "deletedAt": { "type": "Date", "index": true },
  "deletedBy": { "type": "ObjectId", "ref": "User" },
  "deletedByRole": { "type": "string", "enum": ["USER", "ADMIN"], "default": "USER" },

  "createdAt": { "type": "Date", "default": "Date.now" },
  "updatedAt": { "type": "Date", "default": "Date.now" }
}

// Indexes
db.comments.createIndex({ projectId: 1, createdAt: -1 });
db.comments.createIndex({ isHidden: 1, createdAt: -1 });
db.comments.createIndex({ deletedAt: 1, createdAt: -1 });

// RÈGLES MÉTIER (services / routes)
// ============================================================================
//
// **1. Visibilité publique :**
// - Le listing public ne renvoie que les commentaires tels que :
//   - project.status == ACTIVE
//   - project.isArchived == false
//   - isHidden == false
//   - deletedAt non défini
//
// **2. Qui peut commenter :**
// - Uniquement un USER authentifié (pas l’ADMIN, pas un visiteur).
// - Interdit de commenter son propre projet (creatorId != userId).
//
// **3. Suppression par l’auteur :**
// - L’auteur peut supprimer son commentaire → suppression douce :
//   - deletedAt/deletedBy/deletedByRole="USER"
//   - isHidden=true + hiddenReason "Supprimé par l’auteur"
//
// **4. Modération admin :**
// - L’admin peut masquer / réafficher un commentaire avec motif.
// - L’admin peut aussi supprimer via la résolution d’un signalement de commentaire.
//
// **5. Anti-spam :**
// - Rate limiting express-rate-limit (~4 commentaires / minute / utilisateur).
// - Anti-doublon : 2 commentaires identiques consécutifs interdits.
// ============================================================================
