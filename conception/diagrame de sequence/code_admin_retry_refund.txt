@startuml
actor Admin as admin
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "PaymentGateway" as PG

admin -> FE: View failed refunds
FE -> API: GET /api/admin/failed-refunds
API -> DB: Find FailedRefundEvents where resolved = false
DB --> API: list
API --> FE: list
FE --> admin: Show list

admin -> FE: Select event, click "Retry"
FE -> API: POST /api/admin/retry-refund { eventId }

API -> DB: Find FailedRefundEvent
DB --> API: event
API -> DB: Find linked Investment
DB --> API: investment
API -> DB: Find linked Transaction
DB --> API: transaction

' ===== Check transaction state =====
alt investment.status == "SUCCESS" AND transaction.status == "SUCCEEDED" AND transaction.refundStatus IN ["NOT_ATTEMPTED", "FAILED"]
  
  ' ===== Set refundStatus to PENDING before calling provider =====
  API -> DB: Start transaction
  API -> DB: Update Transaction set refundStatus = "PENDING", refundAttempts = refundAttempts + 1, lastRefundAttemptAt = now()
  API -> DB: Commit transaction

  ' Get provider from transaction
  API -> API: provider = transaction.provider
  API -> PG: refundPayment(provider, transaction.providerPaymentId)
  
  alt refund successful
    API -> DB: Start transaction
    API -> DB: Update transaction.status = "REFUNDED", refundStatus = "SUCCEEDED", refundedAt = now()
    API -> DB: Update investment.status = "REFUNDED"
    API -> DB: Update event.resolved = true, resolvedBy = admin.id, resolvedAt = now()
    API -> DB: Insert Notification (in-app) for investor (type PAYMENT_REFUNDED, relatedEntityId = investment.id)
    API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "RETRY_REFUND_SUCCESS", targetType: "FailedRefundEvent", targetId: event.id, details: { provider } }
    API -> DB: Commit transaction
    API --> FE: 200 OK
    FE --> admin: "Refund succeeded"
  else failure
    API -> DB: Start transaction
    API -> DB: Update transaction.refundStatus = "FAILED"
    API -> DB: Increment event.retryCount by 1
    API -> DB: Update event.error = new error
    API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "RETRY_REFUND_FAILED", targetType: "FailedRefundEvent", targetId: event.id, details: { error: event.error, provider } }
    API -> DB: Commit transaction
    API --> FE: 500 "Retry failed"
  end
else
  API -> DB: Update event.resolved = true
  API --> FE: 200 "Already resolved or transaction not in refundable state"
end

note right of API
  - Retry refund uses the same provider as the original transaction (Flouci/Konnect/D17) via PaymentGateway.
  - Before calling provider, refundStatus is set to PENDING and refundAttempts incremented.
  - On success, transaction becomes REFUNDED and investment becomes REFUNDED.
  - On failure, refundStatus reverts to FAILED, retryCount incremented.
end note
@enduml
