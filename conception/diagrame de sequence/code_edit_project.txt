@startuml
actor User as user
participant "Frontend (React)" as FE
participant "Backend API (Node.js)" as API
database "MongoDB" as DB
participant "BullMQ Queue" as Queue

user -> FE: Open project edit form
FE -> API: GET /api/projects/{id}/edit
API -> DB: Find project by id and creatorId (user.id)
DB --> API: project

alt project exists and user is creator
  ' Allow editing for DRAFT, AWAITING_AI, and REJECTED =====
  alt project.status == "DRAFT" or project.status == "AWAITING_AI" or project.status == "REJECTED"
    API --> FE: project data
    FE --> user: Show edit form
    user -> FE: Modify fields (title, desc, goal, deadline, etc.)
    FE -> API: PUT /api/projects/{id} { updated fields }
    API -> API: Validate input (deadline > now(), goal > 0, title/description non‑empty, etc.)
    alt validation fails
      API --> FE: 400 "Validation error: <reason>"
      FE --> user: Show specific error
    else
      API -> DB: Update project fields
      ' If in AWAITING_AI and AI already completed, reset and re‑trigger
      alt project.status == "AWAITING_AI" AND project.aiStatus == "COMPLETED"
        API -> DB: Set project.aiStatus = "PENDING"
        API -> Queue: Add risk analysis job for project
      end
      ' If project was REJECTED, reset to DRAFT (or AWAITING_AI) for re‑submission
      alt project.status == "REJECTED"
        API -> DB: Set project.status = "DRAFT", rejectionReason = null, rejectedBy = null, rejectedAt = null
      end
      API -> DB: Insert AuditLog { actorId: user.id, actorRole: USER, action: "EDIT_PROJECT", ... }
      API --> FE: 200 OK
      FE --> user: "Project updated"
    end
  else
    API --> FE: 403 "Cannot edit project in current status"
    FE --> user: "Project cannot be edited after submission or under review"
  end
else
  API --> FE: 404 "Project not found"
end
@enduml
