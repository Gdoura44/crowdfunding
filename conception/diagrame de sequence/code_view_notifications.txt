@startuml
actor User as user
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB

user -> FE: Request notifications
FE -> API: GET /api/notifications

API -> DB: Find notifications for user, sorted by createdAt desc
DB --> API: list
API --> FE: list

FE --> user: Display notifications

user -> FE: Mark as read
FE -> API: PUT /api/notifications/{id}/read
API -> DB: Update notification.read = true
API --> FE: 200
@enduml
