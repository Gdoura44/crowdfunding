@startuml
actor User as user
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB

user -> FE: Click "Delete Account"
FE -> API: POST /api/users/delete-account
API -> API: Authenticate user

API -> DB: Start MongoDB transaction
API -> DB: Find user by id
DB --> API: user

alt user already deleted or invalid
  API -> API: Abort transaction
  API --> FE: 400 "Account already deleted or cannot delete"
else
  ' ===== 1. Check for pending payouts =====
  API -> DB: Find projects where creatorId = user.id AND status IN ["ACTIVE", "CLOSED"]
  DB --> API: creatorProjects
  loop each project in creatorProjects
    API -> DB: Find payout where projectId = project.id AND status IN ["PENDING", "READY"]
    DB --> API: pendingPayout
    alt pendingPayout exists
      API -> API: Abort transaction
      API --> FE: 400 "Cannot delete account: you have pending payouts. Contact support."
    end
  end
  API -> DB: Find projects where creatorId = user.id AND status IN ["ACTIVE", "APPROVED", "UNDER_REVIEW", "AWAITING_AI"]
  DB --> API: activeProjects
  alt activeProjects not empty
    API -> API: Abort transaction
    API --> FE: 400 "Cannot delete account: you have active projects. Close or suspend them first."
  end

  ' ===== 2. Check for active investments (INITIATED or CANCELLING) =====
  API -> DB: Find investments where investorId = user.id AND status IN ["INITIATED", "CANCELLING"]
  DB --> API: activeInvestments
  alt activeInvestments not empty
    API -> API: Abort transaction
    API --> FE: 400 "Cannot delete account: you have active or cancelling investments. Cancel them first or contact support."
  end

  ' ===== 3. Check that ALL SUCCESS investments are already fully refunded =====
  API -> DB: Find investments where investorId = user.id AND status = "SUCCESS"
  DB --> API: successInvestments
  loop each investment in successInvestments
    API -> DB: Find linked transaction (investmentId = investment.id)
    DB --> API: transaction
    alt transaction.refundStatus != "SUCCEEDED"
      API -> API: Abort transaction
      API --> FE: 400 "Cannot delete account: you have successful investments that are not fully refunded. Wait for refunds to complete or contact support."
    end
  end

  ' ===== 4. Handle projects where user is creator =====
  API -> DB: Find projects where creatorId = user.id
  DB --> API: userProjects
  loop each project in userProjects
  API -> DB: Update project set isCreatorDeleted = true
  ' creatorId remains unchanged (never set to null)
end

  ' Clear bankDetails from payouts where status is not COMPLETED
API -> DB: Find payouts where creatorId = user.id AND status IN ["PENDING","READY","FAILED"]
DB --> API: userPayouts
loop each payout in userPayouts
  API -> DB: Update payout set bankDetails = null   ' privacy requirement for incomplete payouts
end
' For COMPLETED payouts, keep bankDetails for audit (transfer already done)

  ' ===== 5. Anonymize user and mark as deleted =====
  API -> DB: Anonymize user fields (email, name, phone, password, tokens)
  API -> DB: Set isActive = false, deletedAt = now()
  API -> DB: Insert AuditLog { actorId: user.id, actorRole: USER, action: "DELETE_ACCOUNT", targetType: "User", targetId: user.id }
  API -> DB: Insert Notification (in-app) for user (type ACCOUNT_DELETED, relatedEntityType = "USER", relatedEntityId = user.id)
  API -> DB: Commit transaction
  API --> FE: 200 OK
  FE --> user: "Account deleted successfully"
end

note right of API
  **Account Deletion (Synchronous Blocking):**
  - Personal data anonymized immediately.
  - Deletion is **blocked** if:
      • Any pending payout exists (status PENDING/READY)
      • Any active investment exists (status INITIATED or CANCELLING)
      • Any SUCCESS investment exists whose linked transaction has refundStatus != SUCCEEDED
      • Any project they created with status in ["ACTIVE", "APPROVED", "UNDER_REVIEW", "AWAITING_AI", "FUNDED"]
  - For ALL projects: isCreatorDeleted = true, creatorId retained (never set to null).
  - **Payout bank details are cleared for privacy** regardless of payout status.
  - ACCOUNT_DELETED notification created for user.
end note

@enduml
