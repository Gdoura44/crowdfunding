@startuml
actor User as user
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "BullMQ Queue" as Queue

user -> FE: Submit report
FE -> API: POST /api/reports

' ===== NEW: Fetch project to check self-reporting =====
API -> DB: Find project by projectId from request
DB --> API: project

alt user.id == project.creatorId
  API --> FE: 400 "You cannot report your own project"
else
  ' ===== FIXED: Check for duplicate pending OR resolved report of same type =====
  API -> DB: Find existing report by reporterId, projectId, type, status IN ["PENDING", "RESOLVED"]
  DB --> API: existingReport
  alt existingReport exists
    API --> FE: 409 "You already have a pending or resolved report of this type for this project"
  else
    API -> DB: Start MongoDB transaction
    API -> DB: Insert Report (status = PENDING)
    API -> DB: Link report to user & project
    API -> DB: Insert Notification (in-app) for admin (type NEW_REPORT, relatedEntityId = report.id)
    API -> DB: Commit transaction
    API -> Queue: Add job "send-email" with notificationId

    API --> FE: 201 Created
    FE --> user: "Report submitted"
  end
end
