@startuml
actor User as user
participant "Frontend (React)" as FE
participant "Backend API (Node.js)" as API
database "MongoDB" as DB

user -> FE: Select project, click "Archive"
FE -> API: POST /api/projects/{id}/archive

API -> DB: Find project by id and creatorId (user.id)
DB --> API: project

alt project exists and user is creator
  alt project.isArchived == true
    API --> FE: 400 "Project already archived"
  else project.status == "DRAFT" or project.status == "AWAITING_AI" or project.status == "UNDER_REVIEW" or project.status == "REJECTED"
    API -> DB: Update project.isArchived = true
    API -> DB: Insert AuditLog { actorId: user.id, actorRole: USER, action: "ARCHIVE_PROJECT", targetType: "Project", targetId: project.id, details: {} }
    API --> FE: 200 OK
    FE --> user: "Project archived"
  else
    API --> FE: 403 "Cannot archive project in current status"
    FE --> user: "Only draft, awaiting AI, under review, or rejected projects can be archived"
  end
else
  API --> FE: 404 "Project not found"
end
@enduml
