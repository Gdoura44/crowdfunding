@startuml
actor "Investor" as inv
participant "Frontend (React)" as FE
participant "Backend API (Node.js)" as API
database "MongoDB" as DB
participant "PaymentGateway" as PG
participant "BullMQ Queue" as Queue

inv -> FE: View investments, select "Cancel"
FE -> API: POST /api/investments/{id}/cancel

API -> DB: Find investment by id
DB --> API: investment

alt investment exists
  alt investment.status == "INITIATED"
    API -> DB: Start MongoDB transaction
    API -> DB: Find linked transaction (status = "PENDING")
    DB --> API: transaction
    alt transaction exists and transaction.status == "PENDING"
      API -> DB: Update Investment set status = "CANCELLING", cancellingStartedAt = now()
      API -> DB: Commit transaction

      API -> API: provider = transaction.provider
      API -> PG: cancelPayment(provider, transaction.providerPaymentId)
      
      alt cancellation successful
        API -> DB: Start new transaction
        API -> DB: Update Investment set status = "CANCELLED", cancelReason = "USER_REQUESTED", cancelledAt = now(), cancellingStartedAt = null
        API -> DB: Update Transaction set status = "CANCELLED", cancelledAt = now()
        API -> DB: Insert AuditLog { actorId: inv.id, action: "CANCEL_INVESTMENT", targetId: investment.id }
        API -> DB: Recalculate effective total = sum of amounts from investments with status = SUCCESS
              AND linked Transaction.refundStatus IN ["NOT_ATTEMPTED","SUCCEEDED"]
        API -> DB: Find project by projectId
        alt project.status == "FUNDED" AND effective total < project.fundingGoal
        API -> DB: Update project.status = "ACTIVE"
        API -> DB: Insert Notification for creator (type PROJECT_FUNDING_GOAL_LOST)
        else
         ' Remain FUNDED (effective total still >= goal)
        end
        API -> DB: Insert Notification (PAYMENT_CANCELLED)
        API -> DB: Commit transaction
        API -> Queue: Add job "send-email" with notificationId
        API --> FE: 200 OK
        FE --> inv: "Investment cancelled"
        else cancellation failure
          API -> DB: Insert FailedCancellationEvent { investmentId, 
                     reason ="CANCELLATION_FAILED" }
          API -> DB: Insert Notification (CANCELLATION_FAILED)
          API -> Queue: Add job "send-email" with notificationId
          API --> FE: 500 "Cancellation failed, contact support"
          API -> DB: Insert AuditLog { actorId: inv.id, actorRole: USER,
          action: "CANCEL_INVESTMENT_FAILED", targetType: "Investment", targetId: investment.id,
          details: { error, reason: "CANCELLATION_FAILED" } }
        '  Note that investment remains CANCELLING; reconciliation cron will handle stuck states =====
      end
    else
      API --> FE: 400 "Cannot cancel: payment already processed or webhook in progress"
    end

  else investment.status == "SUCCESS"
    API -> DB: Find linked transaction
    DB --> API: transaction
    API -> API: graceExpiry = transaction.createdAt + (cancellationGracePeriodMinutes * 60 * 1000)
    alt now() < graceExpiry
      API -> DB: Start transaction
      API -> DB: Update Investment set status = "CANCELLING", cancellingStartedAt = now()
      API -> DB: Commit transaction
      API -> API: provider = transaction.provider
      API -> PG: cancelPayment(provider, transaction.providerPaymentId)
            alt cancellation successful
        API -> DB: Update Investment -> CANCELLED, cancelReason = "USER_REQUESTED"
        API -> DB: Update Transaction -> CANCELLED
        API -> DB: Recalculate effective total = sum of amounts from investments with status = SUCCESS
              AND linked Transaction.refundStatus IN ["NOT_ATTEMPTED","SUCCEEDED"]
        API -> DB: Find project by projectId
        alt project.status == "FUNDED" AND effective total < project.fundingGoal
          API -> DB: Update project.status = "ACTIVE"
          API -> DB: Insert Notification for creator (type PROJECT_FUNDING_GOAL_LOST)
        else
          ' Remain FUNDED (effective total still >= goal)
        end
        API -> DB: Insert Notification PAYMENT_CANCELLED
        API --> FE: 200 OK
      else cancellation failure
        API -> DB: Insert FailedCancellationEvent { reason = "CANCELLATION_FAILED" }
        API -> DB: Insert Notification CANCELLATION_FAILED
        API --> FE: 500 "Cancellation failed, contact support"
      end
    else
      API --> FE: 400 "Cancellation window expired. Please contact support for refund requests."
    end

  else
    API --> FE: 400 "Cannot cancel: investment not eligible for cancellation"
  end

else
  API --> FE: 404 "Investment not found"
end

note right of API
  **Cancellation flow:**
  - INITIATED → Immediate cancellation attempt.
  - SUCCESS → Grace-period cancellation allowed.
  - Other statuses → Not eligible.
  On failure, a CANCELLATION_FAILED notification is sent.
  **Recovery:** If cancellation fails, investment remains in CANCELLING state.
  The reconciliation cron (every 15 min) will detect CANCELLING >30 min,
  create a FailedCancellationEvent (reason=STUCK_CANCELLING), and set investment.status = FAILED
  for manual admin review.
end note
@enduml
