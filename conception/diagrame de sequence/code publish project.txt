@startuml
actor Admin as admin
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "BullMQ Queue" as Queue

admin -> FE: Publish project
FE -> API: POST /api/admin/projects/{id}/publish

API -> DB: Find project
DB --> API: project

alt project.status == APPROVED
  API -> DB: Start MongoDB transaction
  API -> DB: Update status = ACTIVE, publishedAt = now()
  API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "PUBLISH_PROJECT", targetType: "Project", targetId: project.id, details: {} }
  API -> DB: Insert Notification (in-app) for project owner (type PROJECT_PUBLISHED, relatedEntityId = project.id)
  API -> DB: Commit transaction
  API -> Queue: Add job "send-email" with notificationId
  API --> FE: 200 OK
  FE --> admin: "Project published and open for investment"
else
  API --> FE: 400 "Project not in approved state"
end

note right of API
  **Notifications:** In‑app notification stored in transaction; email queued after commit.
end note
@enduml
