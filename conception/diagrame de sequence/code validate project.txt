@startuml
actor Admin as admin
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "BullMQ Queue" as Queue

admin -> FE: Review project
FE -> API: POST /api/admin/projects/{id}/validate { decision, feedback }

API -> DB: Find project
DB --> API: project

alt project.status == UNDER_REVIEW and project.aiStatus == COMPLETED and aiAnalysis present
  ' Check AI freshness (e.g., analyzedAt within last 30 days)
  alt project.aiAnalysis.analyzedAt < now() - 30 days
    API --> FE: 400 "AI analysis outdated, please retry"
  else project.deadline < now()
    API --> FE: 400 "Project deadline passed"
  else decision == "APPROVED"
    API -> DB: Start MongoDB transaction
    API -> DB: Update status = APPROVED
    API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "VALIDATE_PROJECT", targetType: "Project", targetId: project.id, details: { decision, feedback } }
    API -> DB: Insert Notification (in-app) for project owner (type PROJECT_APPROVED, relatedEntityId = project.id)
    API -> DB: Commit transaction
    API -> Queue: Add job "send-email" with notificationId
    API --> FE: 200 OK
    FE --> admin: "Project approved"
  else decision == "REJECTED"
    ' No investment check needed – project is not yet published
    API -> DB: Start MongoDB transaction
    API -> DB: Update status = REJECTED, rejectionReason = feedback, rejectedBy = admin.id, rejectedAt = now()
    API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "VALIDATE_PROJECT", targetType: "Project", targetId: project.id, details: { decision, feedback } }
    API -> DB: Insert Notification (in-app) for project owner (type PROJECT_REJECTED, relatedEntityId = project.id, include reason in message)
    API -> DB: Commit transaction
    API -> Queue: Add job "send-email" with notificationId
    API --> FE: 200 OK
    FE --> admin: "Project rejected"
  end
else
  API --> FE: 400 "AI analysis not complete or project not ready"
end

note right of API
  **No investment check needed** – project is UNDER_REVIEW, not yet ACTIVE, so no investments can exist.
  **Notifications:** In‑app notification stored in transaction; email queued after commit.
end note
@enduml
