@startuml
actor Admin as admin
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "PaymentGateway" as PG
participant "BullMQ Queue" as Queue

admin -> FE: Select project, click "Deactivate"
FE -> API: POST /api/admin/projects/{id}/deactivate { reason }

API -> DB: Find project by id
DB --> API: project

'  Only ACTIVE projects can be deactivated (SUSPENDED) =====
alt project.status == ACTIVE
  API -> DB: Start MongoDB transaction
  API -> DB: Update project.status = SUSPENDED, rejectionReason = reason, rejectedBy = admin.id, rejectedAt = now()
  
  ' Cancel any pending payout
  API -> DB: Find payout where projectId = id AND status IN ["PENDING", "READY"]
  DB --> API: pendingPayout
  alt pendingPayout exists
    API -> DB: Update payout.status = "CANCELLED", notes = "Project deactivated"
    API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "CANCEL_PAYOUT", targetType: "Payout", targetId: pendingPayout.id, details: { reason } }
    API -> DB: Insert Notification (in-app) for project creator (type PAYOUT_CANCELLED, relatedEntityId: payout.id)
  end
  
  ' Flag investments for background processing
  API -> DB: Find investments with projectId = id AND status = "INITIATED"
  DB --> API: pendingInvestments
  loop each investment in pendingInvestments
    API -> DB: Update Investment set status = "CANCELLING", cancellingStartedAt = now(), scheduledForDeactivation = true where _id = investment._id
  end

  API -> DB: Find investments with projectId = id AND status = "SUCCESS"
  DB --> API: successfulInvestments
  loop each investment in successfulInvestments
    API -> DB: Update Investment set scheduledForDeactivation = true where _id = investment._id
  end

  API -> DB: Insert Notification (in-app) for project creator (type PROJECT_SUSPENDED, include reason)
  API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "DEACTIVATE_PROJECT", targetType: "Project", targetId: project.id, details: { reason } }
  API -> DB: Commit transaction
  API --> FE: 202 Accepted
  FE --> admin: "Project suspension initiated. Refunds and cancellations will be processed in background."

  ' Queue background jobs
  API -> Queue: Add job "send-email" with notificationIds
  API -> Queue: Add job "cancel-pending-investments" with pendingInvestments
  API -> Queue: Add job "refund-successful-investments" with successfulInvestments

else
  API --> FE: 400 "Project cannot be deactivated (only ACTIVE projects can be suspended)"
end

note right of API
  **Background job: cancel-pending-investments**
  For each investment (status = CANCELLING, scheduledForDeactivation = true):
    - Get linked transaction → provider, providerPaymentId
    - PG.cancelPayment(provider, providerPaymentId)
    - On success: update investment → CANCELLED, transaction → CANCELLED, clear scheduledForDeactivation
    - On failure: 
        * investment stays CANCELLING, transaction stays PENDING
        * Insert FailedCancellationEvent { reason: "PROJECT_DEACTIVATED" }
        * Send CANCELLATION_FAILED notification to investor
        * Do NOT clear scheduledForDeactivation (will be retried by cron)

  **Background job: refund-successful-investments**
  For each investment (status = SUCCESS, scheduledForDeactivation = true):
    - Get linked transaction → provider, providerPaymentId
    - PG.refundPayment(provider, providerPaymentId)
    - On success: transaction → REFUNDED, refundStatus = SUCCEEDED, investment → REFUNDED, clear scheduledForDeactivation
    - On failure: 
        * transaction.refundStatus = FAILED, investment stays SUCCESS
        * Insert FailedRefundEvent { reason: "PROJECT_DEACTIVATED" }
        * Send REFUND_FAILED notification to admin
        * scheduledForDeactivation remains true for manual retry

  **Stale flag cleanup:** Cron (cleanup-stuck-states) clears scheduledForDeactivation if operation took >5 min and unresolved.
end note
@enduml
