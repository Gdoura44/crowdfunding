@startuml
actor Investor as inv
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "PaymentGateway" as PG

inv -> FE: View failed investments, click "Retry"
FE -> API: POST /api/investments/{id}/retry

API -> DB: Find original failed investment by id
DB --> API: investment
API -> DB: Count failed investments? Actually we only need to check retry limit
' Instead of counting separate failed investments, check investment.paymentAttempts < maxRetries (e.g., 3)
alt investment.status != "FAILED" OR investment.paymentAttempts >= 4
  API --> FE: 400 "Cannot retry (max 3 retry attempts reached or invalid status)"
else
  API -> DB: Find project, check ACTIVE and deadline
  API -> DB: Find original transaction (any attempt) to get provider
  API -> API: provider = originalTransaction.provider
  API -> API: amount = investment.amount
  API -> API: newIdempotencyKey = new ObjectId()
  API -> API: newAttemptNumber = investment.paymentAttempts + 1

  API -> PG: createPaymentLink(provider, amount, newIdempotencyKey)
  alt provider call fails
    API --> FE: 502 "Payment provider error"
  else
    PG --> API: payment_url, providerPaymentId
    API -> DB: Start transaction
    ' Reuse same Investment: increment paymentAttempts, set status to INITIATED
    API -> DB: Update Investment set paymentAttempts = investment.paymentAttempts + 1, status = "INITIATED", updatedAt = now() where _id = investment._id
    API -> DB: Insert Transaction { investmentId: investment._id, provider, providerPaymentId, amount, status: "PENDING", attemptNumber: newAttemptNumber, createdAt: now() }
    API -> DB: Commit transaction
    alt update/insert success
      API --> FE: payment_url
    else
      API -> PG: cancelPayment(provider, providerPaymentId)
      API --> FE: 409 "Retry conflict"
    end
  end
end
