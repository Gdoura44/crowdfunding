@startuml
actor Admin as admin
participant "Frontend" as FE
participant "Backend API" as API
database "MongoDB" as DB
participant "PaymentGateway" as PG
participant "BullMQ Queue" as Queue

admin -> FE: Select user, click "Deactivate"
FE -> API: PUT /api/admin/users/{id}/deactivate

API -> DB: Find user by id
DB --> API: user

alt user exists and is active
  API -> DB: Start MongoDB transaction
  API -> DB: Find pending investments by user (status = INITIATED)
  DB --> API: pendingInvestments

  loop each investment in pendingInvestments
    API -> DB: Find transaction by investmentId where status = "PENDING"
    DB --> API: transaction
    alt transaction == null
      API -> API: Abort transaction
      API --> FE: 409 "Inconsistent data: investment without pending transaction"
    end
  end

  '  Set scheduledForDeactivation = true for each pending investment =====
  loop each investment in pendingInvestments
    API -> DB: Update Investment set status = "CANCELLING", cancellingStartedAt = now(), scheduledForDeactivation = true
  end

  API -> DB: Update user.isActive = false
  API -> DB: Insert AuditLog { actorId: admin.id, actorRole: ADMIN, action: "DEACTIVATE_USER", targetType: "User", targetId: user.id }
  API -> DB: Insert Notification (in-app) for user (type USER_DEACTIVATED, relatedEntityType = "USER", relatedEntityId = user.id)
  API -> DB: Commit transaction

  API -> Queue: Add job "process-user-deactivation" with { userId, pendingInvestmentIds }
  API -> Queue: Add job "send-email" with notificationId
  API --> FE: 202 Accepted
  FE --> admin: "User deactivation initiated. Cancellations will be processed in background."

  Queue -> API: (consumer) processes job
  loop each investment in pendingInvestments
    API -> DB: Find linked transaction to get provider and providerPaymentId
    DB --> API: transaction
    API -> API: provider = transaction.provider
    API -> PG: cancelPayment(provider, transaction.providerPaymentId)
    alt cancellation successful
      API -> DB: Start transaction
      API -> DB: Update Investment set status = "CANCELLED", cancelReason = "USER_DEACTIVATED", cancelledAt = now(), cancellingStartedAt = null, scheduledForDeactivation = false
      API -> DB: Update Transaction set status = "CANCELLED", cancelledAt = now()
      API -> DB: Commit transaction
    else failure
      ' Investment stays CANCELLING, transaction stays PENDING.
      API -> DB: Insert FailedCancellationEvent { investmentId, error, reason: "CANCELLATION_FAILED", provider, retryCount: 0, resolved: false }
      API -> DB: Insert Notification (in-app) for user (type CANCELLATION_FAILED, relatedEntityId: investment.id, message: "Your pending investment could not be cancelled automatically. Please contact support.")
      API -> Queue: Add job "send-email" with notificationId
      ' scheduledForDeactivation remains true for cron cleanup
      ' ===== Clear the flag immediately to avoid stale state =====
    API -> DB: Update Investment set scheduledForDeactivation = false where _id = investment._id
    end
  end
else
  API --> FE: 404 Not Found
end

note right of API
  **Fixed cancellation failure handling:**
  - Investment remains in `CANCELLING` state.
  - Transaction remains `PENDING`.
  - `scheduledForDeactivation = true` prevents race conditions.
  - `FailedCancellationEvent` created for admin retry.
  - Reconciliation cron will later attempt to resolve stuck `CANCELLING` investments and clear stale flags.
end note
@enduml
