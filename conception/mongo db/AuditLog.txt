{
  "_id": ObjectId,
  
  // actorId + actorRole are the sole authoritative reference for who performed the action
  // adminId field removed entirely to eliminate redundancy
  "actorId": { "type": "ObjectId", "ref": "User", "required": true },
  "actorRole": {
    "type": "string",
    "enum": ["USER", "ADMIN"],
    "required": true
  },
  
  "action": { "type": "string", "required": true },
  // action examples: 
  //   "VALIDATE_PROJECT", "DEACTIVATE_USER", "RETRY_ANALYSIS",
  //   "RESOLVE_REPORT", "APPROVE_PAYOUT", "CANCEL_PAYOUT", 
  //   "REACTIVATE_PROJECT", "ARCHIVE_PROJECT", "DELETE_USER",
  //   "SUBMIT_PROJECT", "CREATE_INVESTMENT", "CANCEL_INVESTMENT"
  
  "targetType": { "type": "string", "required": true },
  // targetType examples: 
  //   "Project", "User", "Report", "Payout", "Investment",
  //   "FailedWorkflowEvent", "FailedCancellationEvent", 
  //   "FailedRefundEvent", "FailedPayoutEvent"
  
  "targetId": { "type": "ObjectId", "required": true },
  
  "details": { "type": "Object" },
  // details: additional context such as:
  //   - decision reason (e.g., "Spam content detected")
  //   - old/new status (e.g., { from: "PENDING", to: "APPROVED" })
  //   - validation notes, resolution outcome, bank details update flag, etc.
  
  "timestamp": { "type": "Date", "default": "Date.now", "required": true }
}

// Indexes
db.auditLogs.createIndex({ actorId: 1, timestamp: -1 });
db.auditLogs.createIndex({ targetId: 1, targetType: 1 });
db.auditLogs.createIndex({ timestamp: -1 });
db.auditLogs.createIndex({ action: 1, timestamp: -1 });

// ============================================================================
// AUDIT LOG USAGE GUIDELINES
// ============================================================================
//
// **When to Create Audit Logs:**
// Audit logs MUST be created for:
// - All admin actions (project validation, user deactivation, report resolution, etc.)
// - All retry operations (failed analysis, refund, cancellation, payout)
// - Sensitive user actions (account deletion, profile updates with PII)
// - Financial operations (payout approval/cancellation, investment creation)
// - Status transitions on critical entities (project publish, suspension, reactivation)
//
// **What NOT to Log:**
// - Read operations (viewing projects, searching, fetching notifications)
// - Normal user browsing behavior
// - Automated system actions (cron jobs) unless they modify critical state
//   (exception: expired project handling SHOULD be logged)
//
// **Actor Reference After User Deletion:**
// - actorId remains valid even after user deletion
// - Points to anonymized User document for audit trail
// - actorRole preserved to indicate whether action was user or admin
// - This enables compliance audits even after users delete accounts
//
// ============================================================================

// ============================================================================
// COMPLIANCE & RETENTION
// ============================================================================
//
// **Retention Policy:**
// - Audit logs MUST be retained for minimum 7 years (financial compliance)
// - Do NOT delete audit logs even after related entities are deleted
// - User deletion anonymizes actorId reference but preserves the log
//
// **Access Control:**
// - Only ADMIN role can view audit logs
// - Audit log viewing is itself NOT logged (to avoid infinite recursion)
// - Export functionality for compliance audits (JSON/CSV)
//
// **Sensitive Data Handling:**
// - Do NOT log PII in details field (no emails, passwords, bank details)
// - Do NOT log full request/response payloads (only relevant fields)
// - Exception: Action type is sufficient to identify what changed (e.g., "UPDATE_BANK_DETAILS")
//
// ============================================================================

// ============================================================================
// SYSTEM CRON USER
// ============================================================================
// A dedicated system user is created during database initialization:
//   _id: ObjectId("000000000000000000000001")  (or a fixed, reserved ID)
//   email: "system@crowdfunding.local"
//   role: "ADMIN"
//   isActive: true
//   (passwordHash and tokens are null/empty - never used for login)
//
// This user is used as actorId for all automated cron jobs and system-initiated
// audit log entries. The ID is hardcoded in code, not generated.
