title UC Annuler une contribution — Annulation / Remboursement (ECB/FSEG)

actor "Utilisateur" as U
boundary "IHM_Contribution" as IHM
control "Controleur_Investissement" as CI
entity "Investissement" as Inv
entity "Transaction" as Tx
entity "Projet" as P
entity "Notification" as N
participant "Passerelle_de_paiement_(demo)" as Pay <<external>>

== Demande d’annulation ==
U -> IHM : demanderAnnulation(investmentId)
IHM -> CI : cancelInvestment(investorId, investmentId)

CI -> Inv : verifierEligibiliteAnnulation()
alt Annulation refusée (délai dépassé / déjà traité)
  CI --> IHM : refuser("Annulation impossible")
  IHM --> U : message d’erreur
else Annulation autorisée
  CI -> Inv : maj(status=CANCELLING)
  CI -> Pay : cancelPayment(providerPaymentId)
  alt Annulation provider KO
    CI -> N : creer(type=CANCELLATION_FAILED, relatedEntity=INVESTMENT)
    CI --> IHM : erreur("Annulation échouée")
    IHM --> U : message d’erreur
  else Annulation provider OK
    alt Paiement confirmé (fenêtre de grâce)
      CI -> Pay : refundPayment(providerPaymentId, amount, reason=USER_REQUESTED)
      alt Remboursement KO
        CI --> IHM : erreur("Remboursement échoué")
        IHM --> U : message d’erreur
      else Remboursement OK
        CI -> P : maj(currentFunding) et normaliser statut
        CI -> Inv : maj(status=REFUNDED, cancelReason=USER_REQUESTED, cancelledAt)
        CI -> Tx : maj(status=REFUNDED, refundStatus=SUCCEEDED, refundedAt)
        CI -> N : creer(type=PAYMENT_REFUNDED, relatedEntity=INVESTMENT)
        opt Objectif perdu (FUNDED -> ACTIVE)
          CI -> N : creer(type=PROJECT_FUNDING_GOAL_LOST, relatedEntity=PROJECT)
        end
      end
    else Paiement non confirmé
      CI -> Inv : maj(status=CANCELLED, cancelReason=USER_REQUESTED, cancelledAt)
      CI -> Tx : maj(status=CANCELLED, cancelledAt)
      CI -> N : creer(type=PAYMENT_CANCELLED, relatedEntity=INVESTMENT)
    end
    CI --> IHM : confirmationAnnulation()
    IHM --> U : afficherConfirmation()
  end
end

@enduml
